// src/app.rs

use eframe::egui;

use crate::bridge::BridgePython;
use crate::config::Config;
use crate::dashboard::{Dashboard, DashboardData};
use crate::export;
use crate::redactor::Redactor;
use crate::widgets::{Sidebar, StatusBar};

pub struct App {
    config: Config,
    bridge: Option<BridgePython>,
    sidebar: Sidebar,
    dashboard: Dashboard,
    redactor: Redactor,
    status: StatusBar,
    error_msg: Option<String>,
    db_loaded: bool,
    pending_action: Option<String>,
    prev_db_path: String,
    group_by: String,
}

impl App {
    pub fn new() -> Self {
        let config = Config::detect();
        let db_path = config
            .default_db
            .as_ref()
            .map(|p| p.display().to_string())
            .unwrap_or_default();

        let mut app = App {
            redactor: Redactor::new(),
            config,
            bridge: None,
            sidebar: Sidebar::new(),
            dashboard: Dashboard::new(),
            status: StatusBar {
                ultima_actualizacion: "--".to_string(),
                total_registros: 0,
            },
            error_msg: None,
            db_loaded: false,
            pending_action: None,
            prev_db_path: String::new(),
            group_by: String::new(),
        };

        let initial_db = db_path.clone();
        app.sidebar.db_path = db_path;
        app.prev_db_path = initial_db;
        app.iniciar_bridge();
        app
    }

    fn iniciar_bridge(&mut self) {
        let python_exe = self.config.python_exe.to_string_lossy().to_string();
        let script = self.config.python_script.to_string_lossy().to_string();

        let bridge = BridgePython::start(&python_exe, &script);
        self.bridge = Some(bridge);

        if !self.sidebar.db_path.is_empty() {
            self.explorar_bd();
        }
    }

    fn explorar_bd(&mut self) {
        let bridge = match &mut self.bridge {
            Some(b) => b,
            None => return,
        };

        if !bridge.is_alive() {
            self.error_msg = Some("Bridge no disponible".to_string());
            return;
        }

        bridge.consultar("explorar", &self.sidebar.db_path, serde_json::json!({}));
        self.dashboard.loading = true;
        self.pending_action = Some("explorar".to_string());
    }

    fn cargar_datos(&mut self) {
        let bridge = match &mut self.bridge {
            Some(b) => b,
            None => return,
        };

        if self.sidebar.db_path.is_empty() || !bridge.is_alive() {
            return;
        }

        self.dashboard.loading = true;
        let filtros = self.sidebar.construir_filtros_json();
        let gb = if self.group_by.is_empty() {
            None
        } else {
            Some(self.group_by.clone())
        };
        bridge.consultar_dashboard(&self.sidebar.db_path, filtros, gb);
        self.pending_action = Some("dashboard".to_string());
    }

    fn poll_bridge(&mut self) {
        let bridge = match &mut self.bridge {
            Some(b) => b,
            None => return,
        };

        if !bridge.is_alive() && bridge.last_error.is_none() {
            bridge.last_error = Some("Proceso Python terminó inesperadamente".to_string());
        }

        let action = match &self.pending_action {
            Some(a) => a.clone(),
            None => return,
        };

        if !bridge.pending {
            self.pending_action = None;
            return;
        }

        if let Some(result) = bridge.poll_response() {
            self.pending_action = None;
            self.dashboard.loading = false;

            match result {
                Ok(data) => match action.as_str() {
                    "explorar" => {
                        match serde_json::from_value::<crate::bridge::SchemaMetadata>(data) {
                            Ok(meta) => {
                                // Pick first categorical as default group_by
                                if self.group_by.is_empty() {
                                    self.group_by = meta
                                        .columnas
                                        .iter()
                                        .find(|c| c.tipo == "categorical")
                                        .map(|c| c.nombre.clone())
                                        .unwrap_or_default();
                                }
                                self.sidebar.set_metadata(meta);
                                self.db_loaded = true;
                                self.error_msg = None;
                                // Auto-load data after exploration
                                self.cargar_datos();
                            }
                            Err(e) => {
                                self.error_msg =
                                    Some(format!("Error parseando schema: {}", e));
                            }
                        }
                    }
                    "dashboard" => {
                        match serde_json::from_value::<DashboardData>(data) {
                            Ok(dd) => {
                                if !dd.grupo_actual.is_empty() {
                                    self.group_by = dd.grupo_actual.clone();
                                }
                                self.dashboard.data = Some(dd);
                                self.status.ultima_actualizacion =
                                    chrono::Local::now().format("%H:%M:%S").to_string();
                                self.error_msg = None;
                            }
                            Err(e) => {
                                self.error_msg =
                                    Some(format!("Error parseando datos: {}", e));
                            }
                        }
                    }
                    _ => {}
                },
                Err(e) => {
                    self.error_msg = Some(format!("Error: {}", e));
                }
            }
        } else {
            // Timeout check: 30 seconds
            self.dashboard.loading = true;
        }
    }

    fn cambiar_db(&mut self, nueva: &str) {
        if nueva.is_empty() {
            return;
        }
        self.sidebar.db_path = nueva.to_string();
        self.db_loaded = false;
        self.dashboard.data = None;
        self.explorar_bd();
    }

    fn exportar_excel(&mut self) {
        let bridge = match &mut self.bridge {
            Some(b) => b,
            None => {
                self.error_msg = Some("Bridge no disponible".to_string());
                return;
            }
        };

        if !bridge.is_alive() {
            self.error_msg = Some("Bridge no disponible".to_string());
            return;
        }

        let filtros = self.sidebar.construir_filtros_json();
        let output_path = self.config.output_dir.join(format!(
            "reporte_{}.xlsx",
            chrono::Local::now().format("%Y%m%d_%H%M%S")
        ));

        bridge.consultar_con_output(
            "exportar_excel",
            &self.sidebar.db_path,
            filtros,
            Some(output_path.to_string_lossy().to_string()),
            None,
            None,
        );
        self.dashboard.loading = true;
        self.pending_action = Some("exportar_excel".to_string());
    }

    fn exportar_pdf(&mut self) {
        let bridge = match &mut self.bridge {
            Some(b) => b,
            None => {
                self.error_msg = Some("Bridge no disponible".to_string());
                return;
            }
        };

        if !bridge.is_alive() {
            self.error_msg = Some("Bridge no disponible".to_string());
            return;
        }

        let filtros = self.sidebar.construir_filtros_json();
        let output_path = self.config.output_dir.join(format!(
            "reporte_{}.pdf",
            chrono::Local::now().format("%Y%m%d_%H%M%S")
        ));

        bridge.consultar_con_output(
            "exportar_pdf",
            &self.sidebar.db_path,
            filtros,
            Some(output_path.to_string_lossy().to_string()),
            None,
            None,
        );
        self.dashboard.loading = true;
        self.pending_action = Some("exportar_pdf".to_string());
    }

    fn exportar_pptx(&mut self) {
        let bridge = match &mut self.bridge {
            Some(b) => b,
            None => {
                self.error_msg = Some("Bridge no disponible".to_string());
                return;
            }
        };

        if !bridge.is_alive() {
            self.error_msg = Some("Bridge no disponible".to_string());
            return;
        }

        let filtros = self.sidebar.construir_filtros_json();
        let output_path = self.config.output_dir.join(format!(
            "reporte_{}.pptx",
            chrono::Local::now().format("%Y%m%d_%H%M%S")
        ));

        bridge.consultar_con_output(
            "exportar_pptx",
            &self.sidebar.db_path,
            filtros,
            Some(output_path.to_string_lossy().to_string()),
            None,
            None,
        );
        self.dashboard.loading = true;
        self.pending_action = Some("exportar_pptx".to_string());
    }
}

impl eframe::App for App {
    fn update(&mut self, ctx: &egui::Context, _frame: &mut eframe::Frame) {
        ctx.set_visuals(egui::Visuals::light());

        // Poll bridge for pending responses
        self.poll_bridge();

        // Check if export completed
        if self.pending_action.is_none() {
            if let Some(ref bridge) = self.bridge {
                if let Some(ref err) = bridge.last_error {
                    self.error_msg = Some(err.clone());
                }
            }
        }

        egui::TopBottomPanel::top("top_bar").show(ctx, |ui| {
            ui.horizontal(|ui| {
                ui.heading("📋 Reporte de Contrataciones");
                if let Some(err) = &self.error_msg {
                    ui.separator();
                    ui.colored_label(egui::Color32::RED, err);
                    if ui.button("✕").clicked() {
                        self.error_msg = None;
                    }
                }
                if self.db_loaded {
                    ui.with_layout(egui::Layout::right_to_left(egui::Align::Center), |ui| {
                        ui.colored_label(egui::Color32::GREEN, "BD cargada");
                        ui.separator();
                        if ui.button("📁 Output")
                            .on_hover_text("Abrir carpeta de archivos generados")
                            .clicked()
                        {
                            self.sidebar.open_output_clicked = true;
                        }
                        if ui.button("📝 Redactar")
                            .on_hover_text("Redactar reporte personalizado con plantillas")
                            .clicked()
                        {
                            self.sidebar.open_redactor_clicked = true;
                        }
                        if ui.button("📊 Excel")
                            .on_hover_text("Exportar a Excel")
                            .clicked()
                        {
                            self.sidebar.export_clicked = true;
                        }
                        if ui.button("📄 PDF")
                            .on_hover_text("Exportar a PDF")
                            .clicked()
                        {
                            self.sidebar.export_pdf_clicked = true;
                        }
                        if ui.button("📽 PPTX")
                            .on_hover_text("Exportar a PowerPoint")
                            .clicked()
                        {
                            self.sidebar.export_pptx_clicked = true;
                        }
                        if ui.button("🔄 Actualizar")
                            .on_hover_text("Recargar datos con filtros actuales")
                            .clicked()
                        {
                            self.sidebar.refresh_clicked = true;
                        }
                    });
                }
            });
        });

        egui::SidePanel::left("sidebar")
            .resizable(true)
            .min_width(180.0)
            .default_width(260.0)
            .show(ctx, |ui| {
                self.sidebar.ui(ui);
            });

        egui::CentralPanel::default().show(ctx, |ui| {
            if self.db_loaded && !self.group_by.is_empty() {
                ui.horizontal(|ui| {
                    ui.label("Agrupar por:");
                    let cat_cols: Vec<String> = self
                        .sidebar
                        .metadata
                        .as_ref()
                        .map(|m| {
                            m.columnas
                                .iter()
                                .filter(|c| c.tipo == "categorical")
                                .map(|c| c.nombre.clone())
                                .collect()
                        })
                        .unwrap_or_default();
                    let mut current = self.group_by.clone();
                    egui::ComboBox::from_id_salt("group_by_selector")
                        .selected_text(crate::dashboard::limpiar_nombre(&current))
                        .show_ui(ui, |ui| {
                            for col in &cat_cols {
                                let label = crate::dashboard::limpiar_nombre(col);
                                if ui
                                    .selectable_value(&mut current, col.clone(), &label)
                                    .clicked()
                                {
                                    if current != self.group_by {
                                        self.group_by = current.clone();
                                        self.cargar_datos();
                                    }
                                }
                            }
                        });
                });
            }
            self.dashboard.ui(ui, &mut self.status);
        });

        egui::TopBottomPanel::bottom("status_bar").show(ctx, |ui| {
            self.status.ui(ui);
        });

        // Detect DB path change (via 📁 button in sidebar)
        let new_path = self.sidebar.db_path.clone();
        if new_path != self.prev_db_path {
            self.cambiar_db(&new_path);
            self.prev_db_path = new_path;
        }

        // Handle button actions
        if self.sidebar.refresh_clicked {
            if self.db_loaded {
                self.cargar_datos();
            } else {
                self.explorar_bd();
            }
            self.sidebar.refresh_clicked = false;
        }

        if self.sidebar.export_clicked {
            self.exportar_excel();
            self.sidebar.export_clicked = false;
        }

        if self.sidebar.export_pdf_clicked {
            self.exportar_pdf();
            self.sidebar.export_pdf_clicked = false;
        }

        if self.sidebar.export_pptx_clicked {
            self.exportar_pptx();
            self.sidebar.export_pptx_clicked = false;
        }

        if self.sidebar.open_redactor_clicked {
            self.redactor.visible = true;
            self.sidebar.open_redactor_clicked = false;
        }

        if self.sidebar.open_output_clicked {
            let _ = export::abrir_carpeta_output(&self.config);
            self.sidebar.open_output_clicked = false;
        }

        // Redactor window
        let data_ref = self.dashboard.data.as_ref();
        let meta_ref = self.sidebar.metadata.as_ref();
        self.redactor.ui(ctx, data_ref, meta_ref);

        // Only repaint when needed
        if self.pending_action.is_some() || self.dashboard.loading {
            ctx.request_repaint();
        }
    }
}

impl Drop for App {
    fn drop(&mut self) {
        if let Some(bridge) = self.bridge.take() {
            bridge.detener();
        }
    }
}

// src/bridge.rs

use serde::Deserialize;
use serde_json::Value;
use std::collections::HashMap;
use std::io::{BufRead, BufReader, BufWriter, Write};
use std::process::{Command, Stdio};
use std::sync::mpsc::{self, Receiver, Sender};
use std::thread::{self, JoinHandle};

enum BridgeCommand {
    Request {
        id: u64,
        tipo: String,
        db: String,
        filtros: Value,
        output: Option<String>,
        vista: Option<String>,
        group_by: Option<String>,
    },
    Quit,
}

#[derive(Debug, Deserialize)]
struct Response {
    #[allow(dead_code)]
    id: u64,
    ok: bool,
    data: Option<Value>,
    error: Option<String>,
}

pub struct BridgePython {
    cmd_tx: Sender<BridgeCommand>,
    resp_rx: Receiver<(u64, Result<Value, String>)>,
    next_id: u64,
    pub pending: bool,
    pub last_error: Option<String>,
    thread: Option<JoinHandle<()>>,
}

impl BridgePython {
    pub fn start(python_exe: &str, script: &str) -> Self {
        let (cmd_tx, cmd_rx) = mpsc::channel();
        let (resp_tx, resp_rx) = mpsc::channel();

        let exe = python_exe.to_string();
        let scr = script.to_string();

        let handle = thread::spawn(move || {
            bg_thread(exe, scr, cmd_rx, resp_tx);
        });

        BridgePython {
            cmd_tx,
            resp_rx,
            next_id: 1,
            pending: false,
            last_error: None,
            thread: Some(handle),
        }
    }

    pub fn consultar(&mut self, tipo: &str, db: &str, filtros: Value) {
        self.consultar_con_output(tipo, db, filtros, None, None, None);
    }

    pub fn consultar_dashboard(
        &mut self,
        db: &str,
        filtros: Value,
        group_by: Option<String>,
    ) {
        self.consultar_con_output("dashboard", db, filtros, None, None, group_by);
    }

    pub fn consultar_con_output(
        &mut self,
        tipo: &str,
        db: &str,
        filtros: Value,
        output: Option<String>,
        vista: Option<String>,
        group_by: Option<String>,
    ) {
        let id = self.next_id;
        self.next_id += 1;

        let ok = self
            .cmd_tx
            .send(BridgeCommand::Request {
                id,
                tipo: tipo.to_string(),
                db: db.to_string(),
                filtros,
                output,
                vista,
                group_by,
            })
            .is_ok();

        if ok {
            self.pending = true;
        }
    }

    /// Non-blocking poll for response. Returns Some once per completed request.
    pub fn poll_response(&mut self) -> Option<Result<Value, String>> {
        match self.resp_rx.try_recv() {
            Ok((_id, result)) => {
                self.pending = false;
                Some(result)
            }
            Err(mpsc::TryRecvError::Empty) => None,
            Err(mpsc::TryRecvError::Disconnected) => {
                self.pending = false;
                self.last_error = Some("Bridge disconnected".to_string());
                None
            }
        }
    }

    pub fn is_alive(&self) -> bool {
        self.thread
            .as_ref()
            .map(|t| !t.is_finished())
            .unwrap_or(false)
    }

    pub fn detener(mut self) {
        let _ = self.cmd_tx.send(BridgeCommand::Quit);
        if let Some(t) = self.thread.take() {
            let _ = t.join();
        }
    }
}

impl Drop for BridgePython {
    fn drop(&mut self) {
        let _ = self.cmd_tx.send(BridgeCommand::Quit);
    }
}

fn bg_thread(
    python_exe: String,
    script: String,
    cmd_rx: Receiver<BridgeCommand>,
    resp_tx: Sender<(u64, Result<Value, String>)>,
) {
    let mut child = match Command::new(&python_exe)
        .arg("-u")
        .arg(&script)
        .stdin(Stdio::piped())
        .stdout(Stdio::piped())
        .stderr(Stdio::piped())
        .spawn()
    {
        Ok(c) => c,
        Err(e) => {
            let _ = resp_tx.send((0, Err(format!("Error spawning Python: {}", e))));
            return;
        }
    };

    let mut stdin = match child.stdin.take() {
        Some(s) => BufWriter::new(s),
        None => {
            let _ = resp_tx.send((0, Err("No stdin".to_string())));
            return;
        }
    };
    let stdout = match child.stdout.take() {
        Some(s) => s,
        None => {
            let _ = resp_tx.send((0, Err("No stdout".to_string())));
            return;
        }
    };

    let mut reader = BufReader::new(stdout);

    loop {
        match cmd_rx.recv() {
            Ok(BridgeCommand::Quit) | Err(_) => break,
            Ok(BridgeCommand::Request {
                id,
                tipo,
                db,
                filtros,
                output,
                vista,
                group_by,
            }) => {
                let request = serde_json::json!({
                    "id": id,
                    "tipo": tipo,
                    "db": db,
                    "filtros": filtros,
                    "output": output,
                    "vista": vista,
                    "group_by": group_by,
                });

                let mut line = match serde_json::to_string(&request) {
                    Ok(s) => s,
                    Err(e) => {
                        let _ = resp_tx.send((id, Err(format!("Serialization: {}", e))));
                        continue;
                    }
                };
                line.push('\n');

                if write_to_stdin(&mut stdin, line.as_bytes()).is_err() {
                    let _ = resp_tx.send((id, Err("Write error: Python died".to_string())));
                    break;
                }

                let mut response_line = String::new();
                if reader.read_line(&mut response_line).is_err() {
                    let _ = resp_tx.send((id, Err("Read error: Python died".to_string())));
                    break;
                }

                match serde_json::from_str::<Response>(&response_line) {
                    Ok(r) if r.ok => {
                        let _ = resp_tx.send((id, Ok(r.data.unwrap_or(Value::Null))));
                    }
                    Ok(r) => {
                        let _ = resp_tx
                            .send((id, Err(r.error.unwrap_or_else(|| "Unknown".to_string()))));
                    }
                    Err(e) => {
                        let _ = resp_tx.send((
                            id,
                            Err(format!("Parse error: {} | raw: {}", e, response_line.trim())),
                        ));
                    }
                }
            }
        }
    }

    let _ = write_to_stdin(&mut stdin, b"{\"tipo\": \"quit\"}\n");
    let _ = child.wait();
}

fn write_to_stdin(w: &mut impl Write, data: &[u8]) -> std::io::Result<()> {
    w.write_all(data)?;
    w.flush()?;
    Ok(())
}

// ─── Schema metadata types ──────────────────────────────────────

#[derive(Debug, Clone, Deserialize)]
pub struct SchemaMetadata {
    #[allow(dead_code)]
    pub vista_principal: String,
    #[allow(dead_code)]
    pub tablas: Vec<String>,
    pub columnas: Vec<ColumnaInfo>,
    #[serde(default)]
    pub dependencias: Vec<DependenciaInfo>,
}

#[derive(Debug, Clone, Deserialize)]
pub struct ColumnaInfo {
    pub nombre: String,
    pub tipo: String,
    #[serde(default)]
    pub valores: Option<Vec<serde_json::Value>>,
    #[serde(default)]
    #[allow(dead_code)]
    pub total_distintos: Option<u64>,
    #[serde(default)]
    pub min: Option<f64>,
    #[serde(default)]
    pub max: Option<f64>,
    #[serde(default)]
    #[allow(dead_code)]
    pub fecha_min: Option<String>,
    #[serde(default)]
    #[allow(dead_code)]
    pub fecha_max: Option<String>,
}

#[derive(Debug, Clone, Deserialize)]
pub struct DependenciaInfo {
    pub columna_padre: String,
    pub columna_dependiente: String,
    pub mapeo: HashMap<String, Vec<String>>,
}

#[derive(Debug, Clone)]
pub enum FiltroValor {
    Categorical { selected: String },
    Date { desde: String, hasta: String },
    Numeric { min: f64, max: f64, active_min: f64, active_max: f64 },
    TextSearch { query: String },
}

impl FiltroValor {
    pub fn to_json(&self) -> Value {
        match self {
            FiltroValor::Categorical { selected } => {
                serde_json::json!({"tipo": "categorical", "valor": selected})
            }
            FiltroValor::Date { desde, hasta } => {
                serde_json::json!({"tipo": "date", "desde": desde, "hasta": hasta})
            }
            FiltroValor::Numeric { active_min, active_max, .. } => {
                serde_json::json!({"tipo": "numeric", "min": active_min, "max": active_max})
            }
            FiltroValor::TextSearch { query } => {
                serde_json::json!({"tipo": "text_search", "valor": query})
            }
        }
    }

    pub fn is_active(&self) -> bool {
        match self {
            FiltroValor::Categorical { selected } => {
                !selected.is_empty() && selected != "__todos__"
            }
            FiltroValor::Date { desde, hasta } => !desde.is_empty() || !hasta.is_empty(),
            FiltroValor::Numeric { active_min, active_max, min, max } => {
                (active_min - min).abs() > 0.001 || (active_max - max).abs() > 0.001
            }
            FiltroValor::TextSearch { query } => !query.is_empty(),
        }
    }

    pub fn reset(&mut self) {
        match self {
            FiltroValor::Categorical { selected } => *selected = "__todos__".to_string(),
            FiltroValor::Date { desde, hasta } => {
                desde.clear();
                hasta.clear();
            }
            FiltroValor::Numeric { active_min, active_max, min, max, .. } => {
                *active_min = *min;
                *active_max = *max;
            }
            FiltroValor::TextSearch { query } => query.clear(),
        }
    }
}

use std::path::{Path, PathBuf};

pub struct Config {
    #[allow(dead_code)]
    pub project_root: PathBuf,
    pub python_exe: PathBuf,
    pub python_script: PathBuf,
    pub default_db: Option<PathBuf>,
    pub output_dir: PathBuf,
}

impl Config {
    pub fn detect() -> Self {
        let project_root = detect_project_root();
        let python_exe = detect_python(&project_root);
        let python_script = project_root.join("python").join("data_service.py");
        let default_db = find_default_db(&project_root);
        let output_dir = project_root.join("output");

        Config {
            project_root,
            python_exe,
            python_script,
            default_db,
            output_dir,
        }
    }
}

fn detect_project_root() -> PathBuf {
    let exe_dir = std::env::current_exe()
        .ok()
        .and_then(|p| p.parent().map(|p| p.to_path_buf()))
        .unwrap_or_else(|| PathBuf::from("."));

    if exe_dir.join("python").join("data_service.py").exists() {
        exe_dir
    } else if PathBuf::from("python/data_service.py").exists() {
        std::env::current_dir().unwrap_or_else(|_| PathBuf::from("."))
    } else {
        exe_dir
    }
}

fn detect_python(project_root: &Path) -> PathBuf {
    let candidates = [
        project_root.join("venv").join("bin").join("python3"),
        project_root.join("venv").join("Scripts").join("python.exe"),
        PathBuf::from("python3"),
        PathBuf::from("python"),
    ];

    for candidate in &candidates {
        if candidate.exists() {
            return candidate.clone();
        }
    }

    PathBuf::from("python3")
}

fn find_default_db(project_root: &Path) -> Option<PathBuf> {
    let data_dir = project_root.join("data");
    if !data_dir.exists() {
        return None;
    }

    std::fs::read_dir(data_dir)
        .ok()?
        .filter_map(|e| e.ok())
        .find(|e| e.path().extension().map(|ext| ext == "db").unwrap_or(false))
        .map(|e| e.path())
}

pub fn limpiar_nombre(n: &str) -> String {

impl Dashboard {
    pub fn new() -> Self {
        Dashboard {
            data: None,
            loading: false,
        }
    }

    pub fn ui(&mut self, ui: &mut egui::Ui, status: &mut StatusBar) {
        if self.loading {
            ui.vertical_centered(|ui| {
                ui.add_space(120.0);
                egui::Spinner::new().ui(ui);
                ui.label("Cargando datos...");
            });
            return;
        }

        let data = match &self.data {
            Some(d) => d,
            None => {
                ui.vertical_centered(|ui| {
                    ui.add_space(120.0);
                    ui.heading("Selecciona una base de datos");
                    ui.label("Usa el botón 📁 en la barra lateral");
                });
                return;
            }
        };

        // Cards responsivas
        let card_w = (ui.available_width() / 3.0).max(120.0).min(220.0);
        ui.horizontal(|ui| {
            metric_card(ui, "Pendientes", data.total_pendientes, egui::Color32::RED, card_w);
            metric_card(ui, "Firmados", data.total_firmados, egui::Color32::GREEN, card_w);
            metric_card(ui, "Total", data.total_general, egui::Color32::BLUE, card_w);
        });

        ui.separator();
        ui.add_space(8.0);

        let has_grupo = !data.por_grupo.is_empty();
        if has_grupo {
            let available = ui.available_width();
            let plot_height = 180.0;
            let plot_width = (available - 16.0) / 2.0;

            // Bar chart (left)
            let sorted_group: BTreeMap<_, _> = data.por_grupo.iter().collect();
            let bars: Vec<Bar> = sorted_group
                .iter()
                .enumerate()
                .map(|(i, (label, count))| {
                    Bar::new(i as f64, **count as f64)
                        .name(label.to_string())
                        .width(0.6)
                        .fill(egui::Color32::from_rgb(66, 133, 244))
                })
                .collect();

            let grupo_label = limpiar_nombre(&data.grupo_actual);
            Plot::new("grupo_plot")
                .height(plot_height)
                .width(plot_width)
                .legend(Legend::default())
                .show(ui, |plot_ui| {
                    plot_ui.bar_chart(BarChart::new(bars).name(&grupo_label));
                });

            // Pie chart (right)
            let (alloc_rect, _) = ui.allocate_exact_size(
                egui::vec2(plot_width, plot_height),
                egui::Sense::hover(),
            );

            let pie_frac = if plot_width < 200.0 { 0.5 } else { 0.55 };
            let pie_right = alloc_rect.left() + alloc_rect.width() * pie_frac;
            let pie_rect = egui::Rect::from_min_max(
                alloc_rect.min,
                egui::pos2(pie_right, alloc_rect.max.y),
            );
            let legend_rect = egui::Rect::from_min_max(
                egui::pos2(pie_right, alloc_rect.min.y),
                alloc_rect.max,
            );

            let painter = ui.painter();
            let center = pie_rect.center();
            let radius = pie_rect.width().min(pie_rect.height()) * 0.35;

            let mut sorted: Vec<_> = data.por_grupo.iter().collect();
            sorted.sort_by(|a, b| a.0.cmp(b.0));
            let total_pie: f64 = sorted.iter().map(|(_, c)| **c as f64).sum();
            let mut start_angle = -std::f32::consts::FRAC_PI_2;

            for (i, (_label, count)) in sorted.iter().enumerate() {
                let fraction = **count as f64 / total_pie;
                let sweep = (fraction * 2.0 * std::f64::consts::PI) as f32;
                let segments = 20.max((sweep * 20.0) as u32);

                let mut points = vec![center];
                for j in 0..=segments {
                    let a = start_angle + sweep * j as f32 / segments as f32;
                    points.push(egui::pos2(
                        center.x + radius * a.cos(),
                        center.y + radius * a.sin(),
                    ));
                }
                painter.add(egui::Shape::Path(egui::epaint::PathShape {
                    points,
                    closed: true,
                    fill: color_for_idx(i),
                    stroke: egui::Stroke::new(1.0, egui::Color32::WHITE).into(),
                }));

                start_angle += sweep;
            }

            painter.circle_stroke(center, radius, egui::Stroke::new(1.0, egui::Color32::GRAY));

            let text_h = 16.0;
            let mut ly = legend_rect.top() + 4.0;
            for (i, (label, count)) in sorted.iter().enumerate() {
                if ly + text_h > legend_rect.bottom() {
                    break;
                }
                painter.rect_filled(
                    egui::Rect::from_min_size(
                        egui::pos2(legend_rect.left() + 2.0, ly + 2.0),
                        egui::vec2(10.0, 10.0),
                    ),
                    2.0,
                    color_for_idx(i),
                );
                let text = format!("{}: {}", label, count);
                painter.text(
                    egui::pos2(legend_rect.left() + 16.0, ly + 7.0),
                    egui::Align2::LEFT_CENTER,
                    text,
                    egui::FontId::proportional(11.0),
                    egui::Color32::DARK_GRAY,
                );
                ly += text_h;
            }
        }

        ui.separator();
        ui.add_space(8.0);

        if !data.columnas_tabla.is_empty() {
            egui::ScrollArea::both()
                .id_salt("tabla_scroll")
                .max_height(300.0)
                .show(ui, |ui| {
                    let columns = &data.columnas_tabla;
                    let cell_w = 100.0_f32.max(ui.available_width() / columns.len().max(1) as f32);

                    egui::Grid::new("tabla_grid")
                        .striped(true)
                        .min_col_width(cell_w)
                        .max_col_width(cell_w * 1.5)
                        .show(ui, |ui| {
                            for col in columns {
                                ui.strong(col.replace('_', " "));
                            }
                            ui.end_row();

                            for row in &data.tabla {
                                for col in columns {
                                    let val = row.get(col).map(|v| match v {
                                        serde_json::Value::String(s) => s.clone(),
                                        serde_json::Value::Number(n) => n.to_string(),
                                        serde_json::Value::Null => String::new(),
                                        _ => format!("{}", v),
                                    });
                                    ui.label(val.unwrap_or_default());
                                }
                                ui.end_row();
                            }
                        });
                });

            status.total_registros = data.total_general;
        }
    }
}

fn metric_card(ui: &mut egui::Ui, label: &str, value: u64, color: egui::Color32, width: f32) {
    let height = 70.0;
    let (rect, _) = ui.allocate_exact_size(egui::vec2(width, height), egui::Sense::hover());

    ui.painter().rect_filled(rect, 6.0, egui::Color32::from_black_alpha(20));
    ui.painter().rect_stroke(
        rect,
        6.0,
        egui::Stroke::new(1.0, color),
        egui::StrokeKind::Middle,
    );

    let text_pos = rect.min + egui::vec2(10.0, 8.0);
    let font_size = width.min(200.0) / 14.0;

    ui.painter().text(
        egui::pos2(text_pos.x, text_pos.y + 6.0),
        egui::Align2::LEFT_TOP,
        label,
        egui::FontId::proportional(font_size.max(10.0)),
        egui::Color32::GRAY,
    );
    ui.painter().text(
        egui::pos2(text_pos.x, text_pos.y + 28.0),
        egui::Align2::LEFT_TOP,
        format!("{}", value),
        egui::FontId::proportional((font_size * 2.0).max(16.0).min(28.0)),
        color,
    );
}

fn color_for_idx(i: usize) -> egui::Color32 {
    const COLORS: [egui::Color32; 10] = [
        egui::Color32::from_rgb(66, 133, 244),
        egui::Color32::from_rgb(219, 68, 55),
        egui::Color32::from_rgb(244, 180, 0),
        egui::Color32::from_rgb(15, 157, 88),
        egui::Color32::from_rgb(171, 71, 188),
        egui::Color32::from_rgb(0, 188, 212),
        egui::Color32::from_rgb(255, 152, 0),
        egui::Color32::from_rgb(96, 125, 139),
        egui::Color32::from_rgb(121, 85, 72),
        egui::Color32::from_rgb(158, 158, 158),
    ];
    COLORS[i % COLORS.len()]
}

use crate::bridge::SchemaMetadata;
use crate::dashboard::DashboardData;

fn limpiar_nombre(n: &str) -> String {
    n.replace('_', " ")
}

pub struct Redactor {
    pub visible: bool,
    pub template: String,
    pub output: String,
}

impl Redactor {
    pub fn new() -> Self {
        Redactor {
            visible: false,
            template: String::new(),
            output: String::new(),
        }
    }

    pub fn redactar(&mut self, data: &DashboardData, meta: &SchemaMetadata) {
        if self.template.trim().is_empty() {
            self.output = "Escribe una plantilla primero.".to_string();
            return;
        }

        let mut result = self.template.clone();

        result = result.replace("#total", &data.total_general.to_string());
        result = result.replace("#pendientes", &data.total_pendientes.to_string());
        result = result.replace("#firmados", &data.total_firmados.to_string());

        let display_cols: Vec<(String, String)> = meta
            .columnas
            .iter()
            .map(|c| (c.nombre.clone(), limpiar_nombre(&c.nombre)))
            .collect();

        // Build a mapping: display_name → column_name
        for (col_name, display_name) in &display_cols {
            let placeholder = format!("#{}", display_name);
            if !result.contains(&placeholder) {
                // Also try with original name
                let alt_placeholder = format!("#{}", col_name);
                if result.contains(&alt_placeholder) {
                    // Already handled by original name
                    continue;
                }
                continue;
            }

            let values: Vec<String> = data
                .tabla
                .iter()
                .map(|row| {
                    row.get(col_name)
                        .map(|v| match v {
                            serde_json::Value::String(s) => s.clone(),
                            serde_json::Value::Number(n) => n.to_string(),
                            _ => format!("{}", v),
                        })
                        .unwrap_or_default()
                })
                .collect();

            let replacement = if values.is_empty() {
                if data.total_general == 0 {
                    "(sin registros)".to_string()
                } else {
                    format!("({} valores)", values.len())
                }
            } else if values.len() == 1 {
                values[0].clone()
            } else {
                let indent = if placeholder.len() > 1 {
                    "  ".to_string()
                } else {
                    String::new()
                };
                let mut lines: Vec<String> = values.iter().map(|v| format!("{}- {}", indent, v)).collect();
                lines.insert(0, format!("({} registros):", values.len()));
                lines.join("\n")
            };

            result = result.replace(&placeholder, &replacement);
        }

        self.output = result;
    }

    pub fn ui(
        &mut self,
        ctx: &egui::Context,
        data: Option<&DashboardData>,
        meta: Option<&SchemaMetadata>,
    ) {
        if !self.visible {
            return;
        }

        let mut open = true;
        egui::Window::new("📝 Redactor de Reportes")
            .open(&mut open)
            .default_size([540.0, 420.0])
            .resizable(true)
            .show(ctx, |ui| {
                ui.vertical(|ui| {
                    ui.label("Escribe tu plantilla. Usa # seguido del nombre de columna visible:");
                    ui.label(
                        "Ej: \"Hay #pendientes documentos pendientes: #contrato\"",
                    );
                    ui.add_space(4.0);

                    // Show available column names
                    if let Some(meta) = meta {
                        if ui.button("📋 Ver columnas disponibles").clicked() {
                            // Toggle via a popup
                        }
                        egui::CollapsingHeader::new("Columnas disponibles")
                            .default_open(false)
                            .show(ui, |ui| {
                                let labels: Vec<String> = meta
                                    .columnas
                                    .iter()
                                    .map(|c| format!("#{}", limpiar_nombre(&c.nombre)))
                                    .collect();
                                let text = labels.join(", ");
                                ui.label(text);
                            });
                    }

                    ui.add_space(4.0);
                    ui.label("Plantilla:");
                    let mut template = self.template.clone();
                    let response = ui.add(
                        egui::TextEdit::multiline(&mut template)
                            .desired_rows(5)
                            .desired_width(f32::INFINITY)
                            .hint_text("Ej: Hay #pendientes documentos de #gerencia"),
                    );
                    if response.changed() {
                        self.template = template;
                    }

                    ui.add_space(8.0);

                    let has_data = data.is_some();
                    let has_meta = meta.is_some();

                    if ui
                        .add_enabled(has_data && has_meta, egui::Button::new("🖊 Redactar"))
                        .clicked()
                    {
                        if let (Some(d), Some(m)) = (data, meta) {
                            self.redactar(d, m);
                        }
                    }

                    ui.add_space(8.0);

                    if !self.output.is_empty() {
                        ui.label("Resultado:");
                        egui::ScrollArea::vertical()
                            .max_height(200.0)
                            .show(ui, |ui| {
                                let mut out = self.output.clone();
                                ui.add(
                                    egui::TextEdit::multiline(&mut out)
                                        .desired_rows(6)
                                        .desired_width(f32::INFINITY)
                                        .font(egui::TextStyle::Monospace)
                                        .interactive(false),
                                );
                            });

                        if ui.button("📋 Copiar al portapapeles").clicked() {
                            ui.ctx().copy_text(self.output.clone());
                        }
                    }
                });
            });

        self.visible = open;
    }
}

use crate::bridge::{ColumnaInfo, FiltroValor, SchemaMetadata};
use eframe::egui;
use std::collections::HashMap;

fn limpiar_nombre(n: &str) -> String {

pub struct Sidebar {
    pub db_path: String,
    pub metadata: Option<SchemaMetadata>,
    pub filtros: HashMap<String, FiltroValor>,
    pub refresh_clicked: bool,
    pub export_clicked: bool,
    pub export_pdf_clicked: bool,
    pub export_pptx_clicked: bool,
    pub open_output_clicked: bool,
    pub open_redactor_clicked: bool,
    pub filtro_order: Vec<String>,
}

impl Sidebar {
    pub fn new() -> Self {
        Sidebar {
            db_path: String::new(),
            metadata: None,
            filtros: HashMap::new(),
            refresh_clicked: false,
            export_clicked: false,
            export_pdf_clicked: false,
            export_pptx_clicked: false,
            open_output_clicked: false,
            open_redactor_clicked: false,
            filtro_order: Vec::new(),
        }
    }

    pub fn set_metadata(&mut self, meta: SchemaMetadata) {
        let mut filtros = HashMap::new();
        let mut order = Vec::new();

        for col in &meta.columnas {
            let filtro = match col.tipo.as_str() {
                "categorical" => FiltroValor::Categorical {
                    selected: "__todos__".to_string(),
                },
                "date" => FiltroValor::Date {
                    desde: String::new(),
                    hasta: String::new(),
                },
                "numeric" => FiltroValor::Numeric {
                    min: col.min.unwrap_or(0.0),
                    max: col.max.unwrap_or(0.0),
                    active_min: col.min.unwrap_or(0.0),
                    active_max: col.max.unwrap_or(0.0),
                },
                "text_search" => FiltroValor::TextSearch {
                    query: String::new(),
                },
                _ => continue,
            };
            filtros.insert(col.nombre.clone(), filtro);
            order.push(col.nombre.clone());
        }

        order.sort();
        self.filtros = filtros;
        self.filtro_order = order;
        self.metadata = Some(meta);
    }

    pub fn construir_filtros_json(&self) -> serde_json::Value {
        let mut map = serde_json::Map::new();
        for (nombre, valor) in &self.filtros {
            map.insert(nombre.clone(), valor.to_json());
        }
        serde_json::Value::Object(map)
    }

    pub fn ui(&mut self, ui: &mut egui::Ui) {
        ui.heading("Filtros");
        ui.separator();

        ui.label("Base de Datos:");
        ui.horizontal(|ui| {
            let path = if self.db_path.is_empty() {
                "Ninguna".to_string()
            } else {
                self.db_path.clone()
            };
            ui.label(path);
            ui.add_space(4.0);
            if ui.button("📁").on_hover_text("Seleccionar archivo .db").clicked() {
                if let Some(path) = rfd::FileDialog::new()
                    .add_filter("SQLite", &["db"])
                    .pick_file()
                {
                    self.db_path = path.display().to_string();
                }
            }
        });

        ui.separator();

        let meta = match &self.metadata {
            Some(m) => m.clone(),
            None => {
                ui.label("Abre una base de datos para ver filtros");
                return;
            }
        };

        let col_map: HashMap<String, ColumnaInfo> = meta
            .columnas
            .iter()
            .map(|c| (c.nombre.clone(), c.clone()))
            .collect();

        let dependientas: Vec<String> = meta
            .dependencias
            .iter()
            .map(|d| d.columna_dependiente.clone())
            .collect();

        let dep_map: HashMap<String, HashMap<String, Vec<String>>> = meta
            .dependencias
            .iter()
            .map(|d| (d.columna_dependiente.clone(), d.mapeo.clone()))
            .collect();

        let mut parent_vals: HashMap<String, Option<String>> = HashMap::new();
        for d in &meta.dependencias {
            if !parent_vals.contains_key(&d.columna_padre) {
                let v = self.filtros.get(&d.columna_padre).and_then(|f| {
                    if let FiltroValor::Categorical { selected } = f {
                        if selected != "__todos__" && !selected.is_empty() {
                            return Some(selected.clone());
                        }
                    }
                    None
                });
                parent_vals.insert(d.columna_padre.clone(), v);
            }
        }

        let filtro_order = self.filtro_order.clone();
        let mut filtros_a_remover = Vec::new();

        egui::ScrollArea::vertical().show(ui, |ui| {
            for col_name in &filtro_order {
                let col_info = match col_map.get(col_name) {
                    Some(c) => c,
                    None => continue,
                };

                let filtro = match self.filtros.get_mut(col_name) {
                    Some(f) => f,
                    None => continue,
                };

                let is_dep = dependientas.contains(col_name);

                ui.horizontal(|ui| {
                    ui.label("≡")
                        .on_hover_text("Arrastrar para reordenar (próximamente)");
                    ui.strong(limpiar_nombre(&col_info.nombre));
                    if filtro.is_active() {
                        if ui
                            .button("✕")
                            .on_hover_text("Limpiar este filtro")
                            .clicked()
                        {
                            filtros_a_remover.push(col_name.clone());
                        }
                    }
                });

                match filtro {
                    FiltroValor::Categorical { selected } => {
                        let all_values: Vec<String> = col_info
                            .valores
                            .as_ref()
                            .map(|vals| {
                                vals.iter()
                                    .filter_map(|x| x.as_str().map(String::from))
                                    .collect()
                            })
                            .unwrap_or_default();

                        let filtered_values: Vec<String> = if is_dep {
                            if let Some(pv) =
                                parent_vals.get(col_name).and_then(|o| o.as_ref())
                            {
                                dep_map
                                    .get(col_name)
                                    .and_then(|m| m.get(pv))
                                    .cloned()
                                    .unwrap_or_default()
                            } else {
                                all_values.clone()
                            }
                        } else {
                            all_values.clone()
                        };

                        let mut items = vec!["__todos__".to_string()];
                        items.extend(filtered_values);

                        let current_label =
                            if selected == "__todos__" || selected.is_empty() {
                                "Todos".to_string()
                            } else {
                                selected.clone()
                            };

                        egui::ComboBox::from_id_salt(format!("cb_{}", col_info.nombre))
                            .selected_text(&current_label)
                            .width(ui.available_width() - 8.0)

#!/usr/bin/env python3
"""
data_service.py - Pipe-based data service for Reporte de Contrataciones.
Protocol: JSON lines via stdin/stdout.
Keeps SQLite connection open for the lifetime of the process.
"""

import json
import sqlite3
import sys
from pathlib import Path

sys.path.insert(0, str(Path(__file__).parent))
import queries
import exporters

_db_conn: sqlite3.Connection | None = None
_db_path: str | None = None


def get_connection(db_path: str) -> sqlite3.Connection:
    global _db_conn, _db_path
    if _db_conn is None or _db_path != db_path:
        if _db_conn is not None:
            try:
                _db_conn.close()
            except Exception:
                pass
        _db_conn = sqlite3.connect(db_path)
        _db_path = db_path
    return _db_conn


def procesar(request: dict) -> dict:
    request_id = request.get("id", 0)
    tipo = request.get("tipo", "")
    db_path = request.get("db", "")
    filtros = request.get("filtros", {})
    output = request.get("output")
    vista = request.get("vista")
    group_by = request.get("group_by")

    if tipo == "quit":
        return {"id": request_id, "ok": True}

    if not db_path:
        return {"id": request_id, "ok": False, "error": "No se especificó ruta de BD"}
    if not Path(db_path).exists():
        return {"id": request_id, "ok": False, "error": f"Archivo no encontrado: {db_path}"}

    try:
        conn = get_connection(db_path)

        if tipo == "explorar":
            data = queries.explorar(conn)
            if "error" in data:
                return {"id": request_id, "ok": False, "error": data["error"]}
            return {"id": request_id, "ok": True, "data": data}

        elif tipo == "dashboard":
            data = queries.dashboard(conn, filtros, vista=vista, group_by=group_by)
            return {"id": request_id, "ok": True, "data": data}

        elif tipo == "exportar_excel":
            if not output:
                return {"id": request_id, "ok": False, "error": "No se especificó ruta de salida"}
            resultado = exporters.excel(conn, filtros, output, vista=vista)
            return {"id": request_id, "ok": True, "data": resultado}

        elif tipo == "exportar_pdf":
            if not output:
                return {"id": request_id, "ok": False, "error": "No se especificó ruta de salida"}
            resultado = exporters.pdf(conn, filtros, output, vista=vista)
            return {"id": request_id, "ok": True, "data": resultado}

        elif tipo == "exportar_pptx":
            if not output:
                return {"id": request_id, "ok": False, "error": "No se especificó ruta de salida"}
            resultado = exporters.pptx(conn, filtros, output, vista=vista)
            return {"id": request_id, "ok": True, "data": resultado}

        else:
            return {"id": request_id, "ok": False, "error": f"Tipo desconocido: {tipo}"}

    except Exception as e:
        return {"id": request_id, "ok": False, "error": str(e)}


def main():
    sys.stdin.reconfigure(encoding="utf-8")
    sys.stdout.reconfigure(encoding="utf-8")

    for line in sys.stdin:
        line = line.strip()
        if not line:
            continue

        try:
            request = json.loads(line)
        except json.JSONDecodeError as e:
            response = {"id": 0, "ok": False, "error": f"JSON inválido: {e}"}
            sys.stdout.write(json.dumps(response, ensure_ascii=False) + "\n")
            sys.stdout.flush()
            continue

        if request.get("tipo") == "quit":
            break

        response = procesar(request)
        sys.stdout.write(json.dumps(response, ensure_ascii=False) + "\n")
        sys.stdout.flush()

    global _db_conn
    if _db_conn is not None:
        try:
            _db_conn.close()
        except Exception:
            pass


if __name__ == "__main__":
    main()

"""
queries.py - Universal data layer for any SQLite schema.
Data-driven column detection. Zero hardcoded naming conventions.
"""

import sqlite3
from typing import Any

import pandas as pd

VISTA_PREDETERMINADA = "vw_reporte_excel_contrataciones"


# ─── EXPLORACION ─────────────────────────────────────────────────


def explorar(conn: sqlite3.Connection) -> dict:
    """Introspecciona la BD y devuelve metadata de columnas para construir filtros.
    Puramente data-driven. Sin naming conventions, sin prefijos hardcodeados.
    """
    tablas = _listar_tablas(conn)
    vista = _encontrar_vista_principal(conn, tablas)
    if not vista:
        vista = _encontrar_tabla_principal(conn, tablas)
    if not vista:
        return {"error": "No se encontró ninguna tabla o vista con datos"}

    cursor = conn.execute(f"PRAGMA table_info({vista})")
    columnas_raw = cursor.fetchall()

    fk_map = _analizar_foreign_keys(conn, tablas)

    columnas = []
    for col in columnas_raw:
        info = _analizar_columna(conn, vista, col, fk_map)
        if info:
            columnas.append(info)

    dependencias = _detectar_dependencias(conn, columnas, fk_map)

    return {
        "vista_principal": vista,
        "tablas": tablas,
        "columnas": columnas,
        "dependencias": dependencias,
    }


def _listar_tablas(conn: sqlite3.Connection) -> list[str]:
    cursor = conn.execute(
        "SELECT name FROM sqlite_master WHERE type='table' ORDER BY name"
    )
    return [r[0] for r in cursor.fetchall()]


def _encontrar_vista_principal(conn: sqlite3.Connection, tablas: list[str]) -> str | None:
    cursor = conn.execute(
        "SELECT name FROM sqlite_master WHERE type='view' ORDER BY name"
    )
    vistas = [r[0] for r in cursor.fetchall()]
    if VISTA_PREDETERMINADA in vistas:
        return VISTA_PREDETERMINADA
    for v in vistas:
        if "reporte" in v.lower() or "vista" in v.lower() or "vw_" in v.lower():
            return v
    return vistas[0] if vistas else None


def _encontrar_tabla_principal(conn: sqlite3.Connection, tablas: list[str]) -> str | None:
    """Heurística data-driven: la tabla con más columnas y más filas suele ser la principal.
    Sin filtros por prefijo de nombre.
    """
    mejor = None
    mejor_puntaje = -1
    for t in tablas:
        cursor = conn.execute(f"PRAGMA table_info({t})")
        cols = cursor.fetchall()
        if len(cols) < 3:
            continue
        cursor.execute(f"SELECT COUNT(*) FROM {t}")
        filas = cursor.fetchone()[0]
        puntaje = len(cols) + (1 if filas > 10 else 0)
        if puntaje > mejor_puntaje:
            mejor_puntaje = puntaje
            mejor = t
    return mejor


def _analizar_columna(
    conn: sqlite3.Connection,
    tabla: str,
    col: tuple,
    fk_map: dict,
) -> dict | None:
    """Analiza una columna y devuelve su metadata para filtros.
    Puramente data-driven. Sin heurísticas de nombre.
    col = (cid, name, type, notnull, dflt_value, pk)
    """
    col_name = col[1]
    col_type = (col[2] or "TEXT").upper()

    if col[5]:  # PK
        return None

    if col_type in ("DATE", "DATETIME", "TIMESTAMP"):
        cursor = conn.execute(f"SELECT MIN({col_name}), MAX({col_name}) FROM {tabla}")
        min_v, max_v = cursor.fetchone()
        return {
            "nombre": col_name,
            "tipo": "date",
            "fecha_min": str(min_v) if min_v else None,
            "fecha_max": str(max_v) if max_v else None,
        }

    if col_type in ("REAL", "FLOAT", "DOUBLE", "NUMERIC", "DECIMAL"):
        cursor = conn.execute(f"SELECT MIN({col_name}), MAX({col_name}) FROM {tabla}")
        min_v, max_v = cursor.fetchone()
        cursor.execute(f"SELECT COUNT(DISTINCT {col_name}) FROM {tabla}")
        distinct = cursor.fetchone()[0]
        if distinct <= 50 and min_v is not None and max_v is not None:
            cursor.execute(
                f"SELECT DISTINCT {col_name} FROM {tabla} ORDER BY {col_name} LIMIT 100"
            )
            values = [r[0] for r in cursor.fetchall() if r[0] is not None]
            return {
                "nombre": col_name,
                "tipo": "categorical",
                "valores": values,
                "total_distintos": distinct,
            }
        return {
            "nombre": col_name,
            "tipo": "numeric",
            "min": float(min_v) if min_v is not None else None,
            "max": float(max_v) if max_v is not None else None,
        }

    if col_type in ("INTEGER", "BIGINT", "SMALLINT", "TINYINT"):
        cursor = conn.execute(f"SELECT COUNT(DISTINCT {col_name}), COUNT(*) FROM {tabla}")
        distinct, total = cursor.fetchone()
        if total > 0 and distinct / total > 0.8:
            return None
        if distinct <= 50:
            cursor.execute(
                f"SELECT DISTINCT {col_name} FROM {tabla} ORDER BY {col_name} LIMIT 100"
            )
            values = [r[0] for r in cursor.fetchall() if r[0] is not None]
            return {
                "nombre": col_name,
                "tipo": "categorical",
                "valores": values,
                "total_distintos": distinct,
            }
        return None

    if col_type in ("TEXT", "VARCHAR", "CHAR"):
        cursor = conn.execute(
            f"SELECT COUNT(DISTINCT {col_name}), AVG(CAST(LENGTH({col_name}) AS REAL)) "
            f"FROM (SELECT {col_name} FROM {tabla} LIMIT 1000)"
        )
        row = cursor.fetchone()
        distinct = row[0] if row else 0
        avg_len = row[1] if row else 0

        if avg_len and avg_len > 80:
            return None

        if distinct <= 50:
            cursor.execute(
                f"SELECT DISTINCT {col_name} FROM {tabla} "
                f"WHERE {col_name} IS NOT NULL AND {col_name} != '' "
                f"ORDER BY {col_name} LIMIT 100"
            )
            values = [r[0] for r in cursor.fetchall()]
            return {
                "nombre": col_name,
                "tipo": "categorical",
                "valores": values,
                "total_distintos": distinct,
            }
        else:
            return {
                "nombre": col_name,
                "tipo": "text_search",
            }

    return None


def _analizar_foreign_keys(conn: sqlite3.Connection, tablas: list[str]) -> dict:
    """Analiza todas las FK de todas las tablas.
    Retorna: { "tabla_origen.columna": {"tabla": destino, "columna": col_dest} }
    """
    fk_map = {}
    for t in tablas:
        cursor = conn.execute(f"PRAGMA foreign_key_list({t})")
        for fk in cursor.fetchall():
            id_fk, seq, tabla_dest, col_origen, col_dest, upd, dele, match = fk
            key = f"{t}.{col_origen}"
            fk_map[key] = {"tabla": tabla_dest, "columna": col_dest}
    return fk_map


def _detectar_dependencias(conn: sqlite3.Connection, columnas: list[dict], fk_map: dict) -> list[dict]:
    """Detecta relaciones de dependencia entre columnas categoricales via FK chains.
    Sin asumir prefijos de tabla. Puramente data-driven.
    """
    dependencias = []
    cat_cols = {c["nombre"]: c for c in columnas if c["tipo"] == "categorical"}

    for hijo_nombre in cat_cols:
        for padre_nombre in cat_cols:
            if padre_nombre == hijo_nombre:
                continue
            mapeo = _construir_mapeo_dependencia(conn, padre_nombre, hijo_nombre, fk_map)
            if mapeo:
                dependencias.append({
                    "columna_padre": padre_nombre,
                    "columna_dependiente": hijo_nombre,
                    "mapeo": mapeo,
                })

    return dependencias


def _construir_mapeo_dependencia(
    conn: sqlite3.Connection,
    col_padre: str,
    col_hijo: str,
    fk_map: dict,
) -> dict | None:
    """Construye mapeo padre→[hijos] via FK chain.
    Busca cadenas FK sin asumir nombres de tabla.
    """
    padre_tables = [k.split(".")[0] for k in fk_map if col_padre.lower() in k.lower()]
    hijo_tables = [k.split(".")[0] for k in fk_map if col_hijo.lower() in k.lower()]

    if not padre_tables or not hijo_tables:
        return None

    for pt in padre_tables:
        for ht in hijo_tables:
            pk_key = f"{pt}.id"
            fk_key = f"{ht}.id"
            fk_info_origen = fk_map.get(fk_key)
            fk_info_pk = fk_map.get(pk_key)

            if not fk_info_origen or not fk_info_pk:
                continue
            if fk_info_origen["tabla"] == fk_info_pk["tabla"]:
                tabla_rel = fk_info_origen["tabla"]
                try:
                    cursor = conn.execute(
                        f"SELECT DISTINCT a.{col_padre}, b.{col_hijo} "
                        f"FROM {tabla_rel} tr "
                        f"JOIN {fk_info_pk['tabla']} a ON tr.{fk_info_pk['columna']} = a.{fk_info_pk.get('columna', 'id')} "
                        f"JOIN {fk_info_origen['tabla']} b ON tr.{fk_info_origen['columna']} = b.{fk_info_origen.get('columna', 'id')} "
                        f"WHERE a.{col_padre} IS NOT NULL AND b.{col_hijo} IS NOT NULL "
                        f"ORDER BY 1, 2"
                    )
                    rows = cursor.fetchall()
                    if not rows:
                        continue
                    mapeo = {}
                    for row in rows:
                        pv = str(row[0])
                        hv = str(row[1])
                        if pv not in mapeo:
                            mapeo[pv] = []
                        if hv not in mapeo[pv]:
                            mapeo[pv].append(hv)
                    if mapeo:
                        return mapeo
                except Exception:
                    continue

    return None

def _cargar_dataframe(conn: sqlite3.Connection, vista: str) -> "pd.DataFrame":
    return pd.read_sql(f"SELECT * FROM {vista}", conn)


def _aplicar_filtros(df: "pd.DataFrame", filtros: dict) -> "pd.DataFrame":
    """Aplica filtros genéricos recibidos desde Rust.
    filtros = { "columna": {"tipo": "...", "valor": ...} }
    """
    for col_name, f_info in filtros.items():
        if col_name not in df.columns:
            continue
        tipo = f_info.get("tipo", "text_search")
        valor = f_info.get("valor")

        if tipo == "categorical" and valor and valor != "__todos__":
            df = df[df[col_name].astype(str) == str(valor)]

        elif tipo == "date":
            desde = f_info.get("desde", "")
            hasta = f_info.get("hasta", "")
            if desde or hasta:
                try:
                    df[col_name] = pd.to_datetime(df[col_name], errors="coerce")
                except Exception:
                    pass
                if desde:
                    df = df[df[col_name] >= desde]
                if hasta:
                    df = df[df[col_name] <= hasta]

        elif tipo == "numeric":
            min_v = f_info.get("min")
            max_v = f_info.get("max")
            if min_v is not None:
                df = df[df[col_name] >= min_v]
            if max_v is not None:
                df = df[df[col_name] <= max_v]

        elif tipo == "text_search" and valor:
            df = df[df[col_name].astype(str).str.contains(str(valor), case=False, na=False)]

    return df


def dashboard(conn: sqlite3.Connection, filtros: dict, vista: str | None = None, group_by: str | None = None) -> dict:
    """Dashboard query: counts, groupings, and detail table."""
    if not vista:
        exp = explorar(conn)
        vista = exp.get("vista_principal", VISTA_PREDETERMINADA)

    df = _cargar_dataframe(conn, vista)
    mapa_metricas = _detectar_metricas(df)
    df = _aplicar_filtros(df, filtros)

    col_estatus = mapa_metricas.get("estatus")

    total_general = len(df)

    pendientes = 0
    firmados = 0
    if col_estatus:
        pendientes = len(df[df[col_estatus].astype(str).str.upper().str.contains("PEND", na=False)])
        firmados = len(df[df[col_estatus].astype(str).str.upper().str.contains("FIRM", na=False)])

    grupo_actual = group_by or mapa_metricas.get("gerencia", "")
    por_grupo = {}
    if grupo_actual and grupo_actual in df.columns:
        for k, v in df.groupby(grupo_actual).size().to_dict().items():
            por_grupo[str(k)] = v

    tabla = df.head(500).fillna("").to_dict(orient="records")
    columnas_tabla = list(df.columns)

    return {
        "total_pendientes": pendientes,
        "total_firmados": firmados,
        "total_general": total_general,
        "por_grupo": por_grupo,
        "grupo_actual": grupo_actual,
        "tabla": tabla,
        "columnas_tabla": columnas_tabla,
    }


def _detectar_metricas(df: "pd.DataFrame") -> dict:
    """Detecta qué columnas usar para las cards y gráficos del dashboard.
    Puramente data-driven. Sin naming conventions.
    """
    mapa = {}
    text_cols = [col for col in df.columns if pd.api.types.is_string_dtype(df[col]) or df[col].dtype == object]

    best_status_col = None
    best_status_score = 0.0

    for col in text_cols:
        values = df[col].astype(str).str.upper()
        ratio_pend = values.str.contains("PEND", na=False).mean()
        ratio_firm = values.str.contains("FIRM", na=False).mean()
        combined = ratio_pend + ratio_firm
        both = ratio_pend > 0.01 and ratio_firm > 0.01

        if combined > 0.5:
            score = combined * (2.0 if both else 1.0)
            if score > best_status_score:
                best_status_score = score
                best_status_col = col

    if best_status_col:
        mapa["estatus"] = best_status_col

    for col in text_cols:
        n = df[col].nunique()
        if 3 < n < 40:
            if "gerencia" not in mapa:
                mapa["gerencia"] = col
            elif "modalidad" not in mapa and col != mapa.get("gerencia"):
                mapa["modalidad"] = col
                break

    return mapa

FOR EACH ROW
BEGIN
    -- 1️⃣ Guardar snapshot del estado ANTERIOR (OLD)
    INSERT INTO historial_expedientes (
        id_expediente, solped, id_gerencia, id_superintendencia,
        id_emisor, id_documento, fecha_presupuesto_base,
        presupuesto_base_usd, tipo_cambio, presupuesto_base_bs,
        id_plan, descripcion_proceso, id_modalidad, id_art,
        id_tipo_contrato, nro_acta_apertura, cantidad_frentes,
        nro_resolucion_jd, id_estatus, id_estado_accion,
        fecha_recibido, fecha_devuelto, id_receptor, nro_proceso,
        id_resultado, nro_contrato_sicac, nro_contrato_sap,
        id_empresa, tiempo_ejecucion, monto_adjudicado_bs,
        monto_adjudicado_usd, fecha_firma_contrato, observaciones_generales
    ) VALUES (
        OLD.id_expediente, OLD.solped, OLD.id_gerencia, OLD.id_superintendencia,
        OLD.id_emisor, OLD.id_documento, OLD.fecha_presupuesto_base,
        OLD.presupuesto_base_usd, OLD.tipo_cambio, OLD.presupuesto_base_bs,
        OLD.id_plan, OLD.descripcion_proceso, OLD.id_modalidad, OLD.id_art,
        OLD.id_tipo_contrato, OLD.nro_acta_apertura, OLD.cantidad_frentes,
        OLD.nro_resolucion_jd, OLD.id_estatus, OLD.id_estado_accion,
        OLD.fecha_recibido, OLD.fecha_devuelto, OLD.id_receptor, OLD.nro_proceso,
        OLD.id_resultado, OLD.nro_contrato_sicac, OLD.nro_contrato_sap,
        OLD.id_empresa, OLD.tiempo_ejecucion, OLD.monto_adjudicado_bs,
        OLD.monto_adjudicado_usd, OLD.fecha_firma_contrato, OLD.observaciones_generales
    );

    -- 2️⃣ Si se QUITÓ la firma -> regresar estatus a PENDIENTE
    UPDATE expedientes
    SET id_estatus = (SELECT id FROM cat_estatus_detalle WHERE nombre = 'PENDIENTE' LIMIT 1)
    WHERE NEW.fecha_firma_contrato IS NULL
      AND OLD.fecha_firma_contrato IS NOT NULL
      AND id_expediente = NEW.id_expediente;

    -- 3️⃣ Actualizar fecha de modificación
    UPDATE expedientes
    SET fecha_actualizacion = CURRENT_DATE
    WHERE id_expediente = NEW.id_expediente;
END;
